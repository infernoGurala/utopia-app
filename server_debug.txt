function doPost(e) {
  var output=ContentService.createTextOutput();
  output.setMimeType(ContentService.MimeType.JSON);
  try {
    var body=JSON.parse(e.postData.contents);
    var rollNumber=(body.rollNumber||'').trim();
    var password=(body.password||'').trim();

    if(!rollNumber||!password){output.setContent(JSON.stringify({error:'Roll number and password are required'}));return output;}

    var BASE='https://info.aec.edu.in', PREFIX='/aus';
    var cookies={};
    var encPassword=encryptPassword(password);

    var loginGet=sendRequest({url:BASE+PREFIX+'/default.aspx',cookies:cookies,followRedirects:true});
    var viewState=extractHiddenInput(loginGet.body,'__VIEWSTATE');
    var viewStateGenerator=extractHiddenInput(loginGet.body,'__VIEWSTATEGENERATOR');
    var eventValidation=extractHiddenInput(loginGet.body,'__EVENTVALIDATION');

    var loginBody='__VIEWSTATE='+encodeURIComponent(viewState)
      +'&__VIEWSTATEGENERATOR='+encodeURIComponent(viewStateGenerator)
      +'&__EVENTVALIDATION='+encodeURIComponent(eventValidation)
      +'&userType=rbtStudent'
      +'&txtUserId='+encodeURIComponent(rollNumber.trim())
      +'&txtPassword='+encodeURIComponent(encPassword)
      +'&hdnpwd='+encodeURIComponent(encPassword)
      +'&btnLogin=LOGIN';

    var loginPost=sendRequest({url:BASE+PREFIX+'/default.aspx',method:'POST',cookies:cookies,body:loginBody,contentType:'application/x-www-form-urlencoded',followRedirects:false,extraHeaders:{'Origin':BASE,'Referer':BASE+PREFIX+'/default.aspx'}});

    var location=loginPost.location||'';
    var redirectedToSM=(loginPost.statusCode>=300&&loginPost.statusCode<400)&&location.toLowerCase().indexOf('studentmaster.aspx')>=0;
    if(!redirectedToSM||!cookies['ASP.NET_SessionId']||!cookies['frmAuth']) throw new Error('Invalid credentials');

    sendRequest({url:BASE+PREFIX+'/StudentMaster.aspx',cookies:cookies,followRedirects:true});
    var profileResponse=sendRequest({url:BASE+PREFIX+'/Academics/StudentProfile.aspx?scrid=17',cookies:cookies,followRedirects:true});

    output.setContent(JSON.stringify({
      debug: 'profile_page',
      status: profileResponse.statusCode,
      html_length: profileResponse.body.length,
      html: profileResponse.body.substring(0, 2000)
    }));
  } catch(err) {
    output.setContent(JSON.stringify({error:err.message}));
  }
  return output;
}

function doGet(e) {
  return ContentService.createTextOutput(JSON.stringify({status:'UTOPIA Attendance Proxy is running'})).setMimeType(ContentService.MimeType.JSON);
}
